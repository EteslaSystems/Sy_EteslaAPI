CREATE PROCEDURE SP_Agregados
(
	IN Opcion INT,
	IN xIdAgregado VARCHAR(255),
	IN xIdPropuesta VARCHAR(255),
	IN Cantidad TINYINT,
	IN Agregado VARCHAR(50),
	IN Costo FLOAT
)
BEGIN
	IF(Opcion = 0) THEN
	/*Insertar*/
	   	INSERT INTO agregados
		VALUES(UUID(), UNHEX(xIdPropuesta), Cantidad, Agregado, Costo);
	ELSEIF(Opcion = 1) THEN
	/*Eliminar*/
	   	UPDATE agregados
		SET deleted_at = NOW()
		WHERE idAgregado = UNHEX(xIdAgregado);
	ELSEIF(Opcion = 2) THEN
	/*Actualizar*/
		UPDATE agregados
		SET tiCantidad = Cantidad, vAgregado = Agregado, fCosto = Costo
		WHERE idAgregado = UNHEX(xIdAgregado);
	ELSEIF(Opcion = 3) THEN
	/*Leer - AGREGADOS pertenecientes a una PROPUESTA*/
		SELECT HEX(idAgregado) AS idAgregado, tiCantidad, vAgregado, fCosto
		FROM agregados
		WHERE id_Propuesta = UNHEX(xIdPropuesta);
	END IF;
END
// DELIMITER ;
