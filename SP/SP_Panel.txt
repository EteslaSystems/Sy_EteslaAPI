DELIMITER //
CREATE PROCEDURE SP_Panel
(
	IN Opcion INT,
	IN xIdPanel VARCHAR(255),
	IN NombreMaterialFot VARCHAR(50),
	IN Marca VARCHAR(50),
	IN Potencia FLOAT,
	IN Precio FLOAT,
	IN Garantia VARCHAR(50),
	IN Origen VARCHAR(100),
	IN ISC FLOAT,
	IN VOC FLOAT,
	IN VMP FLOAT,
    IN ImgRuta VARCHAR(100)
)
BEGIN
    /* Insertar nuevo Panel */
    IF (Opcion = 0) THEN
        INSERT INTO panel
        VALUES (UUID(), NombreMaterialFot, Marca, Potencia, Precio, 'Dolar', Garantia, Origen, ISC, VOC, VMP, ImgRuta, NOW(), NULL, NULL);
    /* Eliminar Panel */
    ELSEIF (Opcion = 1) THEN
        UPDATE panel
        SET deleted_at = NOW()
        WHERE idPanel = UNHEX(xIdPanel);
    /* Editar Panel */
    ELSEIF (Opcion = 2) THEN
        UPDATE panel
        SET vNombreMaterialFot = NombreMaterialFot, vMarca = Marca, fPotencia = Potencia, fPrecio = Precio, vTipoMoneda = 'Dolar', vGarantia = Garantia, vOrigen = Origen, fISC = ISC, fVOC = VOC, fVMP = VMP, updated_at = NOW()
        WHERE idPanel = UNHEX(xIdPanel);
    /* Consultar todos los registros de Paneles */
    ELSEIF (Opcion = 3) THEN
        SELECT HEX(idPanel) AS idPanel, vNombreMaterialFot, vMarca, fPotencia, fPrecio, vTipoMoneda, vGarantia, vOrigen, fISC, fVOC, fVMP, imgRuta, created_at, updated_at
        FROM panel
        WHERE deleted_at IS NULL
        ORDER BY vMarca, fPotencia ASC;
    /* Leer - filtrado */
    ELSEIF (Opcion = 4) THEN
        SELECT HEX(idPanel) AS idPanel, vNombreMaterialFot, vMarca, fPotencia, fPrecio, vTipoMoneda, vGarantia, vOrigen, fISC, fVOC, fVMP, imgRuta, created_at, updated_at
        FROM panel
        WHERE deleted_at IS NULL AND idPanel = UNHEX(xIdPanel);
    END IF;
END
// DELIMITER ;
